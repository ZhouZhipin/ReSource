    bool keep_processing;
    size_t to_copy;
    int substate;
    int state;
    const char *src;
    size_t remaining;
    size_t space;
    size_t consumed;
    int want;
    const void *ptr;
    size_t len;
    char *found;
    bool result;
    size_t buf_sz;
    const void *ref;
    int val;
    size_t blen;
    size_t consumed2;

    if (size == 0)
        return true;
    if (!pp)
        return false;
    if (MHD_str_equal_caseless_n_("application/x-www-form-urlencoded", pp->content_type, 33))
        return post_process_urlencoded(pp, data, size);
    if (!MHD_str_equal_caseless_n_("multipart/form-data", pp->content_type, 19))

            {
                keep_processing = true;
                continue;
            }
            return true;
        }

        remaining = size - total_processed;
        space = buffer_size - buffer_pos;
        to_copy = (remaining < space) ? remaining : space;
        memcpy(buffer + buffer_pos, src, to_copy);
        total_processed += to_copy;
        buffer_pos += to_copy;
        pp->buffer_pos = buffer_pos;

        }

call_process_headers:
        if (process_multipart_headers(pp, &consumed, want))
        {
            goto after_process;
        }
        if (!pp->state)
            return false;
        if (consumed)
        {
            size_t new_pos = pp->buffer_pos - consumed;
            memmove(buffer, buffer + consumed, new_pos);
            pp->buffer_pos = new_pos;
        }
        if (total_processed < size)
        {
            pp->state = 0;
            return false;
        }
        return true;
    }
}

int gz_decomp(gz_state *s)
{
    int start = s->start;
    int ret;
    int err;
    int nread;

    while (1) {
        if (s->have == 0) {
            err = s->errnum;
            if (err && err != -5)
                return -1;
            if (s->flags) {
                gz_error(s, -5, "unexpected end of file");
                s->total_in = (unsigned int)(start - s->start);
                s->next_in = s->next - s->total_in;
                return 0;
            }
            ret = gz_load(s, s->next_out, s->avail_in, &nread);
            if (ret == -1)
                return ret;
            s->have += nread;
            s->next = s->next_out;
            if (s->have == 0) {
                gz_error(s, -5, "unexpected end of file");
                s->total_in = (unsigned int)(start - s->start);
                s->next_in = s->next - s->total_in;
                return 0;
            }
        }
        ret = inflate(&s->strm, 0);
        if (ret == -2 || ret == 2) {
            gz_error(s, -2, "internal error: inflate stream corrupt");
            return -1;
        }
        if (ret == -4) {
            gz_error(s, -4, "out of memory");
            return -1;
        }
        if (ret == -3)
            break;
        err = s->start;
        if (err == 0) {
            s->total_in = (unsigned int)start;
            s->next_in = s->next - s->total_in;
            if (ret == 1)
                s->total_out = 0;
            return 0;
        }
        if (ret == 1) {
            s->total_in = (unsigned int)(start - s->start);
            s->next_in = s->next - s->total_in;
            s->total_out = 0;
            return 0;
        }
    }
    if (!s->msg)
        s->msg = "compressed data error";
    gz_error(s, -3, s->msg);
    return -1;
}

int EVP_PKEY_assign(EVP_PKEY *pkey, int type, void *key)
{
    const EVP_PKEY_ASN1_METHOD *ameth;
    ENGINE *engine;

    if (pkey == NULL) {
        ERR_put_error(6, 158, 156, "p_lib.c", 239);
        return 0;
    }

    if (pkey->pkey.ptr != NULL) {
        if (pkey->ameth && pkey->ameth->pkey_free) {
            pkey->ameth->pkey_free(pkey->pkey.ptr);
            pkey->pkey.ptr = NULL;
        }
        if (pkey->engine) {
            ENGINE_finish(pkey->engine);
            pkey->engine = NULL;
        }
    } else if (type == pkey->type && pkey->ameth != NULL) {
        pkey->pkey.ptr = key;
        return key != NULL;
    } else {
        if (pkey->engine) {
            ENGINE_finish(pkey->engine);
            pkey->engine = NULL;
        }
    }

    ameth = EVP_PKEY_asn1_find(&engine, type);
    if (ameth == NULL) {
        ERR_put_error(6, 158, 156, "p_lib.c", 239);
        return 0;
    }

    pkey->ameth = ameth;
    pkey->type = type;
    pkey->engine = engine;
    pkey->save_type = *(const int *)ameth;
    pkey->pkey.ptr = key;
    return key != NULL;
}

int Curl_resolv(struct connectdata *conn, const char *hostname, int port, bool allowDOH, struct Curl_dns_entry **entry)
{
  struct Curl_easy *data = conn->data;
  struct Curl_dns_entry *dns = NULL;
  int result = CURLE_OK;
  int wait = 0;
  struct Curl_addrinfo *ai = NULL;
  struct in_addr in4;
  struct in6_addr in6;

  *entry = NULL;
  conn->data->state.some_flag &= ~0x20;

  if (data->share && data->share->hostcache)
    Curl_share_lock(data, CURL_LOCK_DATA_DNS, CURL_LOCK_ACCESS_WRITE);

  dns = Curl_cache_lookup(data, hostname, port);
  if (dns) {
    Curl_infof(data, "Hostname %s was found in DNS cache\n", hostname);
    dns->refcount++;
    *entry = dns;
    result = CURLE_OK;
    if (data->share && data->share->hostcache)
      Curl_share_unlock(data, CURL_LOCK_DATA_DNS);
    *entry = dns;
    return result;
  }

  if (data->resolver) {
    Curl_set_in_callback(data, 1);
    result = data->resolver->resolve(data, hostname, port, &wait);
    Curl_set_in_callback(data, 0);
    if (result)
      return result;
  }

  if (inet_pton(AF_INET, hostname, &in4) > 0) {
    ai = Curl_ip2addr(AF_INET, &in4, hostname, port);
    if (ai) {
      if (data->share && data->share->hostcache)
        Curl_share_lock(data, CURL_LOCK_DATA_DNS, CURL_LOCK_ACCESS_WRITE);
      dns = Curl_cache_addr(data, ai, hostname, port);
      if (data->share && data->share->hostcache)
        Curl_share_unlock(data, CURL_LOCK_DATA_DNS);
      if (dns) {
        *entry = dns;
        result = CURLE_OK;
        if (data->share && data->share->hostcache)
          Curl_share_unlock(data, CURL_LOCK_DATA_DNS);
        *entry = dns;
        return result;
      }
      Curl_freeaddrinfo(ai);
      *entry = dns;
      result = CURLE_COULDNT_RESOLVE_HOST;
      if (data->share && data->share->hostcache)
        Curl_share_unlock(data, CURL_LOCK_DATA_DNS);
      *entry = dns;
      return result;
    }
    if (inet_pton(AF_INET6, hostname, &in6) > 0) {
      ai = Curl_ip2addr(AF_INET6, &in6, hostname, port);
      if (ai) {
        if (data->share && data->share->hostcache)
          Curl_share_lock(data, CURL_LOCK_DATA_DNS, CURL_LOCK_ACCESS_WRITE);
        dns = Curl_cache_addr(data, ai, hostname, port);
        if (data->share && data->share->hostcache)
          Curl_share_unlock(data, CURL_LOCK_DATA_DNS);
        if (dns) {
          *entry = dns;
          result = CURLE_OK;
          if (data->share && data->share->hostcache)
            Curl_share_unlock(data, CURL_LOCK_DATA_DNS);
          *entry = dns;
          return result;
        }
        Curl_freeaddrinfo(ai);
        *entry = dns;
        result = CURLE_COULDNT_RESOLVE_HOST;
        if (data->share && data->share->hostcache)
          Curl_share_unlock(data, CURL_LOCK_DATA_DNS);
        *entry = dns;
        return result;
      }
    }
  } else if (inet_pton(AF_INET6, hostname, &in6) > 0) {
    ai = Curl_ip2addr(AF_INET6, &in6, hostname, port);
    if (ai) {
      if (data->share && data->share->hostcache)
        Curl_share_lock(data, CURL_LOCK_DATA_DNS, CURL_LOCK_ACCESS_WRITE);
      dns = Curl_cache_addr(data, ai, hostname, port);
      if (data->share && data->share->hostcache)
        Curl_share_unlock(data, CURL_LOCK_DATA_DNS);
      if (dns) {
        *entry = dns;
        result = CURLE_OK;
        if (data->share && data->share->hostcache)
          Curl_share_unlock(data, CURL_LOCK_DATA_DNS);
        *entry = dns;
        return result;
      }
      Curl_freeaddrinfo(ai);
      *entry = dns;
      result = CURLE_COULDNT_RESOLVE_HOST;
      if (data->share && data->share->hostcache)
        Curl_share_unlock(data, CURL_LOCK_DATA_DNS);
      *entry = dns;
      return result;
    }
  }

  if (Curl_ipvalid(conn)) {
    if (allowDOH && data->state.doh_flag < 0)
      result = Curl_doh(conn, hostname, port, &wait);
    else
      result = Curl_getaddrinfo(conn, hostname, port, &wait);
    if (result) {
      if (data->share && data->share->hostcache)
        Curl_share_unlock(data, CURL_LOCK_DATA_DNS);
      *entry = dns;
      return result;
    }
    if (wait) {
      if (data->state.some_flag & 0x20)
        result = Curl_doh_is_resolved(conn, &dns);
      else
        result = Curl_resolver_is_resolved(conn, &dns);
      if (result) {
        *entry = dns;
        result = dns ? CURLE_OK : CURLE_COULDNT_RESOLVE_HOST;
        if (data->share && data->share->hostcache)
          Curl_share_unlock(data, CURL_LOCK_DATA_DNS);
        *entry = dns;
        return result;
      }
    }
  }
  return CURLE_COULDNT_RESOLVE_HOST;
}

{
    int top = a->top;
    BN_ULONG carry;

    if (top == 0)

    if (w != 0)

            int new_top = top + 1;

                if (bn_expand2(a, new_top) == NULL)
                    return 0;
            }
            a->d[top] = carry;
            a->top = new_top;
            return 1;
        }
        return 1;
    }
    else
    {

int ASN1_BIT_STRING_set_bit(ASN1_BIT_STRING *a, int bit, int value)
{
    int len;
    unsigned char *data;
    int byte_idx;
    unsigned char mask;
    int new_len;
    unsigned char *new_data;

    if (a == NULL || bit < 0)
        return 0;

    a->flags &= ~0x0F;

    len = a->length;
    data = a->data;
    byte_idx = bit >> 3;
    mask = 1 << (7 - (bit & 7));

    if (byte_idx >= len || data == NULL) {
        if (value == 0)
            return 1;

        new_len = byte_idx + 1;
        if (data == NULL) {
            new_data = CRYPTO_malloc(new_len, "a_bitstr.c", 0xC3);
        } else {
            new_data = CRYPTO_realloc_clean(data, len, new_len, "a_bitstr.c", 199);
        }
        if (new_data == NULL) {
            ERR_put_error(0x0D, 183, 65, "a_bitstr.c", 202);
            return 0;
        }
        if (new_len > len)
            memset(new_data + len, 0, new_len - len);
        a->data = new_data;
        a->length = new_len;
        data = new_data;
        len = new_len;
    }

    data[byte_idx] = (data[byte_idx] & ~mask) | (value ? mask : 0);

    for (; len > 0 && data[len - 1] == 0; len--)
        ;

    a->length = len;

    return 1;
}

int read_rangelist(struct comp_unit *unit, struct rangelist **head, bfd_vma offset)
{
    struct arange *arange;
    bfd_vma base, start_offset, end_offset, current_offset, next_offset;
    bfd_vma start_addr, end_addr;
    unsigned char addr_size;
    struct bfd *abfd;
    struct func_table *funcs;
    int signed_addrs;
    struct rangelist *list, *new_node, *prev;

    if (!unit || !head)
        return 0;
    abfd = unit->abfd;
    if (!abfd)
        return 0;
    arange = unit->arange;
    base = unit->base_address;
    start_offset = arange->first;
    end_offset = arange->last;
    if (!start_offset)
    {
        if (!read_section(abfd, unit->debug_info->section_name, arange->section, 0, &start_offset, &end_offset))
            return 0;
        arange = unit->arange;
        start_offset = arange->first;
        end_offset = arange->last;
    }
    if (start_offset + offset < start_offset)
        return 0;
    current_offset = start_offset + offset;
    end_offset += start_offset;
    addr_size = unit->addr_size;

    for (;;)
    {
        if (end_offset < current_offset + (2 * addr_size & 0x1FE))
            return 0;

        funcs = abfd->tdata;
        if (!funcs)
            return 0;

        next_offset = current_offset + addr_size;

        if (end_offset < next_offset)
        {
            start_addr = 0;
        }
        else
        {
            signed_addrs = funcs->signed_addrs;
            if (funcs->type == 5)
            {
                if (signed_addrs)
                {
                    switch (addr_size)
                    {
                        case 4: start_addr = funcs->read_signed_32(current_offset); break;
                        case 8: start_addr = funcs->read_signed_64(current_offset); break;
                        case 2: start_addr = funcs->read_signed_16(current_offset); break;
                        default: bfd_abort(".//dwarf2.c", 0x381, "read_address");
                    }
                }
                else
                {
                    switch (addr_size)
                    {
                        case 4: start_addr = funcs->read_unsigned_32(current_offset); break;
                        case 8: start_addr = funcs->read_unsigned_64(current_offset); break;
                        case 2: start_addr = funcs->read_unsigned_16(current_offset); break;
                        default: bfd_abort(".//dwarf2.c", 0x38F, "read_address");
                    }
                }
                addr_size = unit->addr_size;
                funcs = abfd->tdata;
                next_offset = current_offset + addr_size;
                if (end_offset < next_offset)
                {
                    end_addr = 0;
                    current_offset = next_offset + addr_size;
                    if (!start_addr)
                        return 1;
                    goto process_range;
                }
                signed_addrs = funcs->signed_addrs;
            }
            else
            {
                if (end_offset < current_offset + addr_size)
                {
                    start_addr = 0;
                }
                else
                {
                    switch (addr_size)
                    {
                        case 4: start_addr = funcs->read_unsigned_32(current_offset); break;
                        case 8: start_addr = funcs->read_unsigned_64(current_offset); break;
                        case 2: start_addr = funcs->read_unsigned_16(current_offset); break;
                        default: bfd_abort(".//dwarf2.c", 0x38F, "read_address");
                    }
                }
                addr_size = unit->addr_size;
                funcs = abfd->tdata;
                next_offset = current_offset + addr_size;
            }
        }

        if (end_offset < next_offset + addr_size)
        {
            current_offset = next_offset + addr_size;
            if (!start_addr)
                return 1;
            end_addr = 0;
            goto process_range;
        }

        if (signed_addrs)
        {
            switch (addr_size)
            {
                case 4: end_addr = funcs->read_signed_32(next_offset); break;
                case 8: end_addr = funcs->read_signed_64(next_offset); break;
                case 2: end_addr = funcs->read_signed_16(next_offset); break;
                default: bfd_abort(".//dwarf2.c", 0x381, "read_address");
            }
        }
        else
        {
            switch (addr_size)
            {
                case 4: end_addr = funcs->read_unsigned_32(next_offset); break;
                case 8: end_addr = funcs->read_unsigned_64(next_offset); break;
                case 2: end_addr = funcs->read_unsigned_16(next_offset); break;
                default: bfd_abort(".//dwarf2.c", 0x38F, "read_address");
            }
        }

        current_offset = next_offset + addr_size;
        if (!(end_addr | start_addr))
            return 1;

process_range:
        if (start_addr == (bfd_vma)-1 && end_addr == (bfd_vma)-1)
        {
            base = end_addr;
            continue;
        }

        {
            bfd_vma low = base + start_addr;
            bfd_vma high = base + end_addr;
            if (low == high)
                continue;

            list = *head;
            if (list)
            {
                while (list)
                {
                    if (high == list->high)
                    {
                        list->low = low;
                        addr_size = unit->addr_size;
                        goto next_iteration;
                    }
                    if (low == list->low)
                    {
                        list->low = low;
                        addr_size = unit->addr_size;
                        goto next_iteration;
                    }
                    prev = list;
                    list = list->next;
                }
            }
            else
            {
                list = *head;
            }

            new_node = bfd_zalloc(abfd, sizeof(struct rangelist));
            if (!new_node)
                return 0;

            new_node->low = low;
            new_node->high = high;
            new_node->next = *head;
            *head = new_node;
            addr_size = unit->addr_size;
        }
    }

next_iteration:
    ;
}

    BIO_printf(bp, "Private key: ");
    nid = EVP_PKEY_base_id(pkey);
    if (nid == 811 || nid == 812) {
        ec = EVP_PKEY_get0(pkey);
        if (ec) {
            priv_key = EC_KEY_get0_private_key(ec);
            if (priv_key) {
                BN_print(bp, priv_key);
            } else {
                BIO_printf(bp, "<undefined)");
            }
        } else {
            BIO_printf(bp, "<undefined)");
        }
    } else {
        BIO_printf(bp, "<undefined)");
    }
    BIO_printf(bp, "\n");
    ctx = BN_CTX_new();
    if (!ctx) {
        ERR_GOST_error(131, 65, "gost_ameth.c", 462, line);
        return 0;
    }
    BN_CTX_start(ctx);
    x = BN_CTX_get(ctx);
    y = BN_CTX_get(ctx);
    ec = EVP_PKEY_get0(pkey);
    pub_key = EC_KEY_get0_public_key(ec);
    ec = EVP_PKEY_get0(pkey);
    group = EC_KEY_get0_group(ec);
    if (EC_POINT_get_affine_coordinates_GFp(group, pub_key, x, y, ctx)) {
        if (BIO_indent(bp, indent, 128)) {
            BIO_printf(bp, "Public key:\n");
            if (BIO_indent(bp, indent + 3, 128)) {
                BIO_printf(bp, "X:");
                BN_print(bp, x);
                BIO_printf(bp, "\n");
                BIO_indent(bp, indent + 3, 128);
                BIO_printf(bp, "Y:");
                BN_print(bp, y);
                BIO_printf(bp, "\n");
                BN_CTX_end(ctx);
                BN_CTX_free(ctx);
                ec = EVP_PKEY_get0(pkey);
                group = EC_KEY_get0_group(ec);
                nid = EC_GROUP_get_curve_name(group);
                if (BIO_indent(bp, indent, 128)) {
                    name = OBJ_nid2ln(nid);
                    BIO_printf(bp, "Parameter set: %s\n", name);
                    return 1;
                }
            }
        }
        return 0;
    }
    ERR_GOST_error(131, 16, "gost_ameth.c", 472, line);
    BN_CTX_free(ctx);
    return 0;
}

    {
        sqlite3_log(21, "API call with %s database connection pointer", "NULL");
        return 21;
    }
    else if (*(int*)((char*)db + 112) != 0xA129C0F7)
    {
        int magic = *(int*)((char*)db + 112);
        if (magic == 1266094736 || magic == -264537850)
            sqlite3_log(21, "API call with %s database connection pointer", "unopened");
        else
            sqlite3_log(21, "API call with %s database connection pointer", "invalid");
        return 21;
    }
    if (!zSql)
    {
        sqlite3_log(21, "API call with %s database connection pointer", "NULL");
        return 21;
    }
    else
    {
        return sqlite3LockAndPrepare(db, zSql, nBytes, 0, ppStmt, pzTail);
    }
}

CURLcode Curl_pretransfer(struct Curl_easy *data)
{
    CURLcode result;
    struct connectdata *conn = data->conn;
    CURLU *url_handle = data->state.url_handle;
    int v5;
    size_t upload_size;

    if (!conn || (data->state.conn_used & 1)) {
        if (data->state.conn_used & 1) {
            Curl_cfree(conn);
            data->state.conn_used &= ~1;
            data->conn = NULL;
        }
        if (!url_handle) {
            Curl_failf(data, "No URL set!");
            return CURLE_URL_MALFORMAT;
        }
        if (curl_url_get(url_handle, 0, data->state.url_buffer, 0)) {
            Curl_failf(data, "No URL set!");
            return CURLE_URL_MALFORMAT;
        }
    }

    v5 = data->set.httpreq;
    data->state.offset = 0;
    data->state.keepon = 0;
    Curl_cfree(data->state.recv_buf);
    data->state.recv_buf = NULL;
    data->state.httpreq = v5;
    data->conn = (void *)data->state.url_buffer;
    data->state.wildcard_flags = ((data->state.some_byte >> 2) << 7) | (data->state.wildcard_flags & 0x55);
    data->state.upload_fromhere = data->state.in_buffer;
    data->state.upload_fromhere_backup = data->state.in_buffer_backup;

    if (v5 == 5)
        data->state.upload_size = data->set.postdata;
    else if (v5 == 1 || v5 == 6)
        data->state.upload_size = 0;
    else {
        upload_size = data->set.optval;
        data->state.upload_size = upload_size;
        if (upload_size == (size_t)-1 && data->set.optstring)
            data->state.upload_size = strlen(data->set.optstring);
    }

    if (data->state.cookie_load)
        Curl_cookie_loadfiles(data);

    if (!data->state.host_pairs || (result = Curl_loadhostpairs(data), !result)) {
        data->state.wildcard_flags |= 0x10;
        Curl_initinfo(data);
        Curl_pgrsResetTransferSizes(data);
        Curl_pgrsStartNow(data);
        data->state.bytecount &= data->state.upload_fromhere;
        data->state.bytecount_backup &= data->state.upload_fromhere_backup;

        if ((signed char)data->state.wildcard_flags >= 0 || data->state.wildcard_init_done || !Curl_wildcard_init(&data->state.wildcard_state))
            result = 0;
        else
            result = CURLE_READ_ERROR;
    }

    return result;
}

struct asection *
bfd_make_section_old_way(struct bfd *abfd, const char *name)
{
    struct asection *sec;
    struct asection *prev;
    int count;

    if ((abfd->flags & 0x20) != 0)
    {
        bfd_set_error(5);
        return NULL;
    }

    if (strcmp(name, "*ABS*") == 0)
    {
        sec = &bfd_abs_section;
        if (abfd->xvec->bfd_section_from_bfd(abfd, sec))
            return sec;
        return NULL;
    }
    else if (strcmp(name, "*COM*") == 0)
    {
        sec = (struct asection *)&bfd_std_section;
        if (abfd->xvec->bfd_section_from_bfd(abfd, sec))
            return sec;
        return NULL;
    }
    else if (strcmp(name, "*UND*") == 0)
    {
        sec = &bfd_und_section;
        if (abfd->xvec->bfd_section_from_bfd(abfd, sec))
            return sec;
        return NULL;
    }
    else if (strcmp(name, "*IND*") == 0)
    {
        sec = &bfd_ind_section;
        if (abfd->xvec->bfd_section_from_bfd(abfd, sec))
            return sec;
        return NULL;
    }
    else
    {
        sec = (struct asection *)bfd_hash_lookup(abfd->section_htab, name, 1, 0);
        if (sec == NULL)
            return NULL;

        if (sec->name != NULL)
            return sec;

        sec->name = name;
        sec->id = bfd_section_id;
        count = abfd->section_count;
        sec->owner = abfd;
        sec->index = count;

        if (!abfd->xvec->bfd_section_from_bfd(abfd, sec))
            return NULL;

        abfd->section_count++;
        sec->prev = NULL;
        prev = abfd->section_last;
        bfd_section_id++;
        if (prev != NULL)
        {
            sec->prev = prev;
            prev->next = sec;
        }
        else
        {
            abfd->section_head = sec;
        }
        abfd->section_last = sec;

        return sec;
    }
}

int MHD_queue_auth_fail_response(struct MHD_Connection *connection, const char *realm, const char *opaque, struct MHD_Response *response, int stale)
{
  struct MHD_Daemon *daemon;
  char nonce[130];
  int sec;
  size_t nonce_len;
  unsigned int nonce_count;
  int hash;
  int idx;
  struct MHD_NonceEntry *entry;
  const char *stale_str;
  int header_len;
  char *header;
  int ret;
  __int64 nonce_params[2];
  int v35[2];
  char v44;
  __int64 v45;

  daemon = connection->daemon;
  sec = MHD_monotonic_sec_counter();
  nonce_params[0] = 16;
  nonce_params[1] = (__int64)&v44;
  v35[0] = daemon->nonce_seed;
  calculate_nonce(sec, (char *)daemon->nonce_seed, daemon->nonce_size, daemon->nonce_key, (char *)v35, realm, (__int64)nonce_params, nonce);
  nonce_len = strlen(nonce) + 1;
  if (nonce_len > 129 || (nonce_count = daemon->nonce_count) == 0)
  {
    MHD_DLOG(daemon, "Could not register nonce (is the nonce array size zero?).\n");
    return 0;
  }
  if (nonce[0])
  {
    unsigned int v17 = 0;
    int v15 = nonce[0];
    for (size_t i = 0; i < nonce_len - 1; i++)
    {
      unsigned int v18 = v17;
      int v19 = v15 ^ (v17 >> 24);
      v15 = nonce[i + 1];
      v17 = (v18 << 8) | v19;
    }
    idx = (v17 % nonce_count) * 152;
  }
  else
  {
    idx = 0;
  }
  entry = (struct MHD_NonceEntry *)(daemon->nonce_array + idx);
  if (pthread_mutex_lock(&daemon->nonce_mutex))
  {
    mhd_panic(mhd_panic_cls, "digestauth.c", 560, "Failed to lock mutex.\n");
    return 0;
  }
  memcpy(entry->nonce, nonce, nonce_len);
  entry->timestamp = 0;
  entry->count = 0;
  if (pthread_mutex_unlock(&daemon->nonce_mutex))
  {
    mhd_panic(mhd_panic_cls, "digestauth.c", 571, "Failed to unlock mutex.\n");
    return 0;
  }
  stale_str = stale ? ",stale=\"true\"" : "";
  header_len = snprintf(NULL, 0, "Digest realm=\"%s\",qop=\"auth\",nonce=\"%s\",opaque=\"%s\",algorithm=md5%s", realm, nonce, opaque, stale_str);
  if (header_len <= 0)
  {
    MHD_DLOG(daemon, "Failed to add Digest auth header.\n");
    return 0;
  }
  header = calloc(1, header_len + 1);
  if (!header)
  {
    MHD_DLOG(daemon, "Failed to allocate memory for auth response header.\n");
    return 0;
  }
  if (snprintf(header, header_len + 1, "Digest realm=\"%s\",qop=\"auth\",nonce=\"%s\",opaque=\"%s\",algorithm=md5%s", realm, nonce, opaque, stale_str) != header_len)
  {
    free(header);
    MHD_DLOG(daemon, "Failed to add Digest auth header.\n");
    return 0;
  }
  ret = MHD_add_response_header(response, "WWW-Authenticate", header);
  free(header);
  if (ret == 1)
    return MHD_queue_response(connection, 401, response);
  MHD_DLOG(daemon, "Failed to add Digest auth header.\n");
  return 0;
}

_BYTE *ReadBlobString(Image *image, BYTE *string)
{
  BYTE *p;
  size_t count;
  int ch;

  if (!image)
    return NULL;
  if (image->signature != MagickCoreSignature)
    __assert_fail("image->signature == MagickCoreSignature", "MagickCore/blob.c", 0x12E8u, "ReadBlobString");
  p = string;
  count = 0;
  while ((ch = ReadBlobByte(image)) != -1)
  {
    *p = ch;
    if (ch == '\n')
    {
      if (count && string[count - 1] == '\r')
        p = &string[count - 1];
      else
        p = &string[count];
      *p = '\0';
      return string;
    }
    p++;
    count++;
    if (count == 4095)
    {
      string[4095] = '\0';
      return string;
    }
  }
  if (count)
  {
    string[count] = '\0';
    return string;
  }
  return NULL;
}

bfd *bfd_openw(const char *filename, const char *target)
{
    bfd *abfd;
    struct objalloc *memory;
    int id;
    size_t len;
    char *name;

    abfd = bfd_zmalloc(sizeof(bfd));
    if (abfd == NULL)
        return NULL;

    if (bfd_use_reserved_id)
    {
        --bfd_use_reserved_id;
        id = --bfd_reserved_id_counter;
    }
    else
    {
        id = bfd_id_counter++;
    }
    abfd->id = id;

    memory = objalloc_create();
    if (memory == NULL)
    {
        bfd_set_error(6);
        free(abfd);
        return NULL;
    }
    abfd->memory = memory;

    abfd->arch = &bfd_default_arch_struct;

    if (!bfd_hash_table_init_n(&abfd->section_hash, bfd_section_hash_newfunc, sizeof(struct bfd_section), 13))
    {
        objalloc_free(memory);
        free(abfd);
        return NULL;
    }

    abfd->target = bfd_find_target(target);
    if (abfd->target == NULL)
    {
        bfd_hash_table_free(&abfd->section_hash);
        objalloc_free(memory);
        free(abfd);
        return NULL;
    }

    len = strlen(filename) + 1;
    name = bfd_malloc(len);
    if (name == NULL)
    {
        abfd->filename = NULL;
        bfd_hash_table_free(&abfd->section_hash);
        objalloc_free(memory);
        free(abfd);
        return NULL;
    }
    abfd->filename = memcpy(name, filename, len);
    abfd->flags = (abfd->flags & 0xE7) | 0x10;

    if (!bfd_open_file(abfd))
    {
        bfd_set_error(1);
        free(abfd->filename);
        bfd_hash_table_free(&abfd->section_hash);
        objalloc_free(memory);
        free(abfd);
        return NULL;
    }

    return abfd;
}

int ec_GFp_mont_group_set_curve(EC_GROUP *group, const BIGNUM *p, const BIGNUM *a, const BIGNUM *b, BN_CTX *ctx)
{
    BN_MONT_CTX *mont = NULL;
    BIGNUM *R = NULL;
    BN_CTX *tmp_ctx = NULL;
    int ret = 0;

    if (group->mont)
    {
        BN_MONT_CTX_free(group->mont);
        group->mont = NULL;
    }
    if (group->R)
    {
        BN_free(group->R);
        group->R = NULL;
    }

    if (ctx == NULL)
    {
        tmp_ctx = BN_CTX_new();
        if (tmp_ctx == NULL)
            return 0;
        ctx = tmp_ctx;
    }

    do {
        mont = BN_MONT_CTX_new();
        if (mont == NULL)
            break;
        if (!BN_MONT_CTX_set(mont, p, ctx))
        {
            ERR_put_error(16, 189, 3, "ecp_mont.c", 233);
            break;
        }
        R = BN_new();
        if (R == NULL)
            break;
        if (!BN_mod_mul_montgomery(R, BN_value_one(), &mont->RR, mont, ctx))
            break;

        group->mont = mont;
        group->R = R;

        ret = ec_GFp_simple_group_set_curve(group, p, a, b, ctx);
        if (!ret)
        {
            BN_MONT_CTX_free(group->mont);
            group->mont = NULL;
            BN_free(group->R);
            group->R = NULL;
        }
    } while (0);

    if (tmp_ctx)
        BN_CTX_free(tmp_ctx);
    return ret;
}

            return NULL;
        }
        if (a) *a = ret;

    ret->group = group;

    return ret;
}

    if (value == ((void *)0))
        return ((void *)0);
    entry_type = *(int *)value;
    payload = *(void **)((char *)value + 8);
    switch (type)
    {
        case RegistryImageList:
            if (entry_type != RegistryImageList)
                return ((void *)0);
            result = CloneImageInfo(payload);
            break;
        case RegistryString:
            switch (entry_type)
            {
                case RegistryImageList:
                    result = ConstantString((void *)((char *)payload + 8816));
                    break;
                case RegistryString:
                    result = ConstantString(payload);
                    break;
                case RegistryImageInfo:
                    result = ConstantString((void *)((char *)payload + 1016));
                    break;
                default:
                    return ((void *)0);
            }
            break;
        case RegistryImageInfo:
            if (entry_type == RegistryImageInfo)
                result = CloneImageList(payload, extra);
            else
                return ((void *)0);
            break;
        default:
            return ((void *)0);
    }
    return result;
}

unsigned __int64 *monty_pow(void *ctx, unsigned __int64 *base, unsigned __int64 *exponent)
{
    unsigned __int64 base_len = *base;
    unsigned __int64 *base_data = (unsigned __int64 *)base[1];
    unsigned __int64 *mod_ptr = *(unsigned __int64 **)((char *)ctx + 40);
    unsigned __int64 mod_len = *mod_ptr;
    unsigned __int64 *mod_data = (unsigned __int64 *)mod_ptr[1];
    unsigned __int64 mod_data_size = 8 * mod_len;
    unsigned __int64 ctx_size = *(__int64 *)((char *)ctx + 16);
    unsigned __int64 accum_len = 2 * ctx_size;
    unsigned __int64 accum_data_size = 16 * ctx_size;

    if (!base_len || !mod_len || !accum_len)
    {
        mp_make_sized_part_5(1LL, 16LL, 0, 0, 0, 0);
        return 0;
    }

    unsigned __int64 *base_copy = (unsigned __int64 *)safemalloc(1uLL, 0x10uLL, 8 * base_len);
    *base_copy = base_len;
    base_copy[1] = (unsigned __int64)(base_copy + 2);
    smemclr(base_copy + 2, 8 * base_len);
    memcpy((void *)base_copy[1], base_data, 8 * base_len);

    unsigned __int64 *mod_copy = (unsigned __int64 *)safemalloc(1uLL, 0x10uLL, mod_data_size);
    *mod_copy = mod_len;
    mod_copy[1] = (unsigned __int64)(mod_copy + 2);
    smemclr(mod_copy + 2, mod_data_size);
    memcpy((void *)mod_copy[1], mod_data, 8 * mod_len);

    unsigned __int64 *accum = (unsigned __int64 *)safemalloc(1uLL, 0x10uLL, accum_data_size);
    *accum = accum_len;
    accum[1] = (unsigned __int64)(accum + 2);
    smemclr(accum + 2, accum_data_size);

    unsigned __int64 bit_index;
    unsigned __int64 exp_len = *exponent;
    unsigned __int64 *exp_data = (unsigned __int64 *)exponent[1];

    for (bit_index = 0; bit_index < (exp_len << 6); bit_index++)
    {
        unsigned __int64 max_size = *base_copy;
        if (*mod_copy >= *base_copy)
            max_size = *mod_copy;
        if (max_size > *accum)
            max_size = *accum;

        unsigned __int64 tmp_limbs = 6 * max_size;
        unsigned __int64 tmp_bytes = 48 * max_size;
        unsigned __int64 *tmp_mul = (unsigned __int64 *)safemalloc(1uLL, 0x10uLL, tmp_bytes);
        if (!tmp_limbs)
        {
            mp_make_sized_part_5(1LL, 16LL, 0, 0, 0, 0);
            return 0;
        }
        *tmp_mul = tmp_limbs;
        tmp_mul[1] = (unsigned __int64)(tmp_mul + 2);
        smemclr(tmp_mul + 2, tmp_bytes);

        mp_mul_internal(accum, mod_copy, base_copy, *tmp_mul, tmp_mul[1], 0);
        smemclr(tmp_mul[1], 8 * *tmp_mul);
        smemclr(tmp_mul, 16uLL);
        safefree(tmp_mul);

        unsigned __int64 reduced_len = monty_reduce_internal(
            (unsigned __int64 *)ctx,
            accum,
            **(unsigned __int64 **)((char *)ctx + 64),
            *(unsigned __int64 *)(*(unsigned __int64 *)((char *)ctx + 64) + 8uLL),
            0, 0);

        unsigned __int64 copy_len = *accum;
        if (*accum >= reduced_len)
            copy_len = reduced_len;
        unsigned __int64 copy_bytes = 8 * copy_len;
        memmove((void *)accum[1], (const void *)accum[1], copy_bytes);
        smemclr(accum[1] + copy_bytes, 8 * (*accum - copy_len));
        smemclr(*(unsigned __int64 *)(*(unsigned __int64 *)((char *)ctx + 64) + 8uLL), 8 * **(unsigned __int64 **)((char *)ctx + 64));

        unsigned __int64 bit_mask;
        if (bit_index >> 6 < exp_len)
            bit_mask = -((exp_data[bit_index >> 6] >> (bit_index & 0x3F)) & 1uLL);
        else
            bit_mask = 0;

        unsigned __int64 mod_limbs = *mod_copy;
        if (mod_limbs)
        {
            unsigned __int64 ac_limbs = *accum;
            unsigned __int64 mod_data_ptr = mod_copy[1];
            unsigned __int64 ac_data_ptr = accum[1];
            unsigned __int64 max_i = mod_limbs;
            if (mod_limbs > ac_limbs)
                max_i = ac_limbs;
            unsigned __int64 i;
            for (i = 0; i < max_i; i++)
                *(unsigned __int64 *)(mod_data_ptr + 8 * i) ^= bit_mask & (*(unsigned __int64 *)(mod_data_ptr + 8 * i) ^ *(unsigned __int64 *)(ac_data_ptr + 8 * i));

            if (mod_limbs > i)
            {
                unsigned __int64 start = i;
                unsigned __int64 len = mod_limbs - i;
                unsigned __int64 aligned_start = (mod_data_ptr + 8 * i) >> 3 & 1;
                if (len > 1 && mod_limbs >= start + 1)
                {
                    if (aligned_start)
                    {
                        *(unsigned __int64 *)(mod_data_ptr + 8 * i) &= ~bit_mask;
                        i = start;
                    }
                    else
                    {
                        i = start;
                    }
                    unsigned __int64 quad_count = len - aligned_start;
                    unsigned __int64 *ptr = (unsigned __int64 *)(mod_data_ptr + 8 * (aligned_start + start));
                    __m128i mask_vec = _mm_loadl_epi64((const __m128i *)&bit_mask);
                    mask_vec = _mm_xor_si128(_mm_unpacklo_epi64(mask_vec, mask_vec), (__m128i)xmmword_B3A10);
                    for (unsigned __int64 q = 0; q < (quad_count >> 1); q++)
                        *(__m128i *)(ptr + q * 2) = _mm_and_si128(_mm_load_si128((const __m128i *)(ptr + q * 2)), mask_vec);
                    i = start + (quad_count & ~1uLL);
                    if (quad_count == (quad_count & ~1uLL))
                        goto end_of_conditional;
                    i = start + (quad_count & ~1uLL);
                }
                *(unsigned __int64 *)(mod_data_ptr + 8 * i) &= ~bit_mask;
                if (mod_limbs > i + 1)
                    *(unsigned __int64 *)(mod_data_ptr + 8 * (i + 1)) &= ~bit_mask;
            }
        }
end_of_conditional:

        unsigned __int64 sq_max = *base_copy;
        if (*accum <= *base_copy)
            sq_max = *accum;
        unsigned __int64 sq_tmp_limbs = 6 * sq_max;
        unsigned __int64 sq_tmp_bytes = 48 * sq_max;
        unsigned __int64 *tmp_sq = (unsigned __int64 *)safemalloc(1uLL, 0x10uLL, sq_tmp_bytes);
        if (!sq_tmp_limbs)
        {
            mp_make_sized_part_5(1LL, 16LL, 0, 0, 0, 0);
            return 0;
        }
        *tmp_sq = sq_tmp_limbs;
        tmp_sq[1] = (unsigned __int64)(tmp_sq + 2);
        smemclr(tmp_sq + 2, sq_tmp_bytes);

        mp_mul_internal(accum, base_copy, base_copy, *tmp_sq, tmp_sq[1], 0);
        smemclr(tmp_sq[1], 8 * *tmp_sq);
        smemclr(tmp_sq, 16uLL);
        safefree(tmp_sq);

        unsigned __int64 sq_reduced = monty_reduce_internal(
            (unsigned __int64 *)ctx,
            accum,
            **(unsigned __int64 **)((char *)ctx + 64),
            *(unsigned __int64 *)(*(unsigned __int64 *)((char *)ctx + 64) + 8uLL),
            0, 0);

        unsigned __int64 sq_copy_len = *accum;
        if (*accum >= sq_reduced)
            sq_copy_len = sq_reduced;
        unsigned __int64 sq_copy_bytes = 8 * sq_copy_len;
        memmove((void *)accum[1], (const void *)accum[1], sq_copy_bytes);
        smemclr(accum[1] + sq_copy_bytes, 8 * (*accum - sq_copy_len));
        smemclr(*(unsigned __int64 *)(*(unsigned __int64 *)((char *)ctx + 64) + 8uLL), 8 * **(unsigned __int64 **)((char *)ctx + 64));

        unsigned __int64 bc_len = *base_copy;
        if (*accum <= *base_copy)
            bc_len = *accum;
        unsigned __int64 bc_bytes = 8 * bc_len;
        memmove((void *)base_copy[1], (const void *)accum[1], bc_bytes);
        smemclr(base_copy[1] + bc_bytes, 8 * (*base_copy - bc_len));
    }

    smemclr(base_copy[1], 8 * *base_copy);
    smemclr(base_copy, 16uLL);
    safefree(base_copy);
    smemclr(accum[1], 8 * *accum);
    smemclr(accum, 16uLL);
    safefree(accum);
    smemclr(*(unsigned __int64 *)(*(unsigned __int64 *)((char *)ctx + 64) + 8uLL), 8 * **(unsigned __int64 **)((char *)ctx + 64));
    return mod_copy;
}

int compare(void *a1, unsigned char **a2)
{
    unsigned char *first_str = *(unsigned char **)a1;
    size_t first_len = *((size_t *)a1 + 1);
    unsigned char *second_str = a2[0];
    unsigned char *second_end = a2[1];
    size_t second_len;
    int result;

    if (!keylist || (keycompare((unsigned char **)a1, a2) == 0 && !unique && !stable))
    {
        if (second_end == NULL)
            second_len = 0;
        else
            second_len = second_end - second_str;

        if (first_len == 1)
        {
            result = -(second_end != NULL);
        }
        else
        {
            result = 1;
            if (second_end != NULL)
            {
                size_t n = second_len;
                size_t first_n = first_len - 1;

                if (first_n <= second_len)
                    n = first_n;

                if (hard_LC_COLLATE)
                {
                    result = xmemcoll0(first_str, first_len, second_str, second_len);
                }
                else
                {
                    result = memcmp(first_str, second_str, n);
                    if (result == 0)
                    {
                        if (first_len - 1 < second_len)
                            result = -1;
                        else if (first_len - 1 > second_len)
                            result = 1;
                        else
                            result = 0;
                    }
                }
            }
        }

        if (reverse)
            result = -result;
    }
    else
    {
        result = keycompare((unsigned char **)a1, a2);
    }

    return result;
}

pid_t pipe_fork(int fds[2], int max_retries)
{
  if (rpl_pipe2(fds) < 0)
    return -1;
  if (nprocs > nmerge + 1)
    reap_constprop_18();
  do
  {
    int status;
    pid_t waited = waitpid(-1, &status, 1);
    if (waited < 0)
    {
      close(fds[0]);
      close(fds[1]);
      return -1;
    }
    if (waited == 0)
      break;
    void *entry = hash_delete(proctab, &waited);
    if (!entry)
      break;
    *((char *)entry + 12) = 2;
    if (status & 0x7f || (status >> 8) & 0xff)
    {
      char *prog = quotearg_style(4, compress_program);
      char *msg = dcgettext(0, "%s [-d] terminated abnormally", 5);
      error(2, 0, msg, prog);
    }
  } while (nprocs > 0);
  double sleep_time = 0.25;
  while (1)
  {
    sigset_t oldmask;
    pthread_sigmask(0, &caught_signals, &oldmask);
    void *saved_temphead = temphead;
    temphead = 0;
    pid_t pid = fork();
    temphead = saved_temphead;
    pthread_sigmask(2, &oldmask, 0);
    if (pid < 0)
    {
      int saved_errno = errno;
      if (saved_errno == 11)
      {
        xnanosleep(sleep_time);
        sleep_time *= 2;
        do
        {
          int st;
          pid_t w = waitpid(-1, &st, 1);
          if (w < 0)
          {
            close(fds[0]);
            close(fds[1]);
            return -1;
          }
          if (w == 0)
            break;
          void *e = hash_delete(proctab, &w);
          if (!e)
            break;
          *((char *)e + 12) = 2;
          if (st & 0x7f || (st >> 8) & 0xff)
          {
            char *prog = quotearg_style(4, compress_program);
            char *msg = dcgettext(0, "%s [-d] terminated abnormally", 5);
            error(2, 0, msg, prog);
          }
        } while (nprocs > 0);
        if (--max_retries == 0)
        {
          close(fds[0]);
          close(fds[1]);
          errno = saved_errno;
          return -1;
        }
        continue;
      }
      else
      {
        close(fds[0]);
        close(fds[1]);
        errno = saved_errno;
        return -1;
      }
    }
    if (pid == 0)
    {
      close(0);
      close(1);
      return 0;
    }
    else
    {
      ++nprocs;
      return pid;
    }
  }
}

int ihex_object_p(bfd *abfd)
{
  char *p;
  int c;
  unsigned int line_count;
  unsigned long long data_allocated;
  unsigned char *data;
  unsigned int byte_count;
  unsigned long long base_address;
  unsigned long long extended_address;
  unsigned long long start_address;
  bfd_section *section;
  bfd_section *current_section;
  long long saved_section_ptr;
  long long record_offset;
  char header[9];
  char err_char[2];
  unsigned char ch;
  int v;

  if (!inited_10969)
  {
    inited_10969 = 1;
    hex_init();
  }

  if (bfd_seek(abfd, 0LL, 0LL) != 0)
    return 0;

  if (bfd_bread(header, 9, abfd) != 9)
  {
    if (bfd_get_error() != 18)
      return 0;
    bfd_set_error(3);
    return 0;
  }

  if (header[0] != ':')
  {
    bfd_set_error(3);
    return 0;
  }

  if (hex_value[(unsigned char)header[1]] == 99 ||
      hex_value[(unsigned char)header[2]] == 99 ||
      hex_value[(unsigned char)header[3]] == 99 ||
      hex_value[(unsigned char)header[4]] == 99 ||
      hex_value[(unsigned char)header[5]] == 99 ||
      hex_value[(unsigned char)header[6]] == 99)
  {
    bfd_set_error(3);
    return 0;
  }

  int v2 = hex_value[(unsigned char)header[7]];
  if (v2 == 99)
  {
    bfd_set_error(3);
    return 0;
  }

  int v3 = hex_value[(unsigned char)header[8]];
  if (v3 == 99 || (unsigned int)(v3 + 16 * v2) > 5)
  {
    bfd_set_error(3);
    return 0;
  }

  saved_section_ptr = abfd[31];
  bfd_section *sec = bfd_alloc(abfd, 16);
  if (!sec || (abfd[31] = (long long)sec, sec->vma = 0, sec->size = 0, bfd_seek(abfd, 0LL, 0LL) != 0))
  {
    long long current = abfd[31];
    if (current != saved_section_ptr)
    {
      if (current)
        bfd_release(abfd, current);
    }
    abfd[31] = saved_section_ptr;
    return 0;
  }

  abfd[21] = 0;
  data_allocated = 0;
  data = 0;
  line_count = 1;
  current_section = 0;
  start_address = 0;
  base_address = 0;
  extended_address = 0;

  int error = 0;
  while (!error && bfd_bread(&ch, 1, abfd) == 1)
  {
    if (ch == 13)
      continue;
    if (ch == 10)
    {
      line_count++;
      continue;
    }
    if (ch != ':')
    {
      if ((sch_istable[ch] & 0x10) == 0)
      {
        error = 1;
        break;
      }
      err_char[0] = ch;
      err_char[1] = 0;
      bfd_error_handler(abfd, dcgettext("bfd", "%pB:%d: unexpected character `%s' in Intel Hex file", 5), line_count, err_char);
      bfd_set_error(0x11);
      error = 1;
      break;
    }

    record_offset = bfd_tell(abfd);
    unsigned char record_buf[8];
    if (bfd_bread(record_buf, 8, abfd) != 8)
    {
      error = 1;
      break;
    }

    unsigned char *rp = record_buf;
    int hex_digit;
    for (int i = 0; i < 8; i++)
    {
      hex_digit = hex_value[rp[i]];
      if (hex_digit == 99)
      {
        if ((sch_istable[rp[i]] & 0x10) == 0)
        {
          error = 1;
          break;
        }
        err_char[0] = rp[i];
        err_char[1] = 0;
        bfd_error_handler(abfd, dcgettext("bfd", "%pB:%d: unexpected character `%s' in Intel Hex file", 5), line_count, err_char);
        bfd_set_error(0x11);
        error = 1;
        break;
      }
    }
    if (error) break;

    byte_count = 16 * hex_value[record_buf[0]] + hex_value[record_buf[1]];
    unsigned long long needed = 2 * byte_count + 2;
    if (needed > data_allocated)
    {
      data = bfd_realloc(data, needed);
      if (!data)
      {
        error = 1;
        break;
      }
      data_allocated = needed;
    }

    if (bfd_bread(data, needed, abfd) != needed)
    {
      error = 1;
      break;
    }

    unsigned char *dp = data;
    for (unsigned int i = 0; i < 2 * byte_count; i++)
    {
      if (hex_value[dp[i]] == 99)
      {
        if ((sch_istable[dp[i]] & 0x10) == 0)
        {
          error = 1;
          break;
        }
        err_char[0] = dp[i];
        err_char[1] = 0;
        bfd_error_handler(abfd, dcgettext("bfd", "%pB:%d: unexpected character `%s' in Intel Hex file", 5), line_count, err_char);
        bfd_set_error(0x11);
        error = 1;
        break;
      }
    }
    if (error) break;

    unsigned long address = hex_value[record_buf[3]] + 16 * hex_value[record_buf[2]] + ((16 * hex_value[record_buf[1]] + hex_value[record_buf[0]]) << 8);
    unsigned int rec_type = hex_value[record_buf[7]] + 16 * hex_value[record_buf[6]];
    unsigned int checksum = 0;
    checksum = byte_count + (address & 0xFF) + ((address >> 8) & 0xFF) + rec_type;
    for (unsigned int i = 0; i < byte_count; i++)
    {
      checksum += 16 * hex_value[data[2*i]] + hex_value[data[2*i+1]];
    }
    checksum = (unsigned char)-checksum;
    unsigned char expected = 16 * hex_value[data[2*byte_count]] + hex_value[data[2*byte_count+1]];
    if (checksum != expected)
    {
      bfd_error_handler(abfd, dcgettext("bfd", "%pB:%u: bad checksum in Intel Hex file (expected %u, found %u)", 5), line_count, checksum, expected);
      bfd_set_error(0x11);
      error = 1;
      break;
    }

    switch (rec_type)
    {
      case 0:
        if (current_section && (long long *)current_section->vma == (long long *)(address + base_address + start_address))
        {
          current_section->size += byte_count;
        }
        else if (byte_count)
        {
          snprintf(err_char, sizeof(err_char), "%c", data[0]);
          p = bfd_alloc(abfd, strlen(err_char) + 1);
          if (!p)
          {
            error = 1;
            break;
          }
          strcpy(p, err_char);
          current_section = bfd_make_section_with_flags(abfd, p, 259);
          if (!current_section)
          {
            error = 1;
            break;
          }
          current_section->vma = (long long *)(base_address + start_address + address);
          current_section->size = byte_count;
          current_section->filepos = record_offset - 1;
        }
        continue;

      case 1:
        if (!abfd[21])
          abfd[21] = address;
        error = 1;
        break;

      case 2:
        if (byte_count != 2)
        {
          bfd_error_handler(abfd, dcgettext("bfd", "%pB:%u: bad extended address record length in Intel Hex file", 5), line_count);
          bfd_set_error(0x11);
          error = 1;
          break;
        }
        current_section = 0;
        base_address = 16 * (hex_value[data[3]] + 16 * hex_value[data[2]] + ((hex_value[data[1]] + 16 * hex_value[data[0]]) << 8));
        continue;

      case 3:
        if (byte_count != 4)
        {
          bfd_error_handler(abfd, dcgettext("bfd", "%pB:%u: bad extended start address length in Intel Hex file", 5), line_count);
          bfd_set_error(0x11);
          error = 1;
          break;
        }
        current_section = 0;
        abfd[21] += 16 * (hex_value[data[3]] + 16 * hex_value[data[2]] + ((hex_value[data[1]] + 16 * hex_value[data[0]]) << 8))
                    + (hex_value[data[7]] + 16 * hex_value[data[6]] + ((hex_value[data[5]] + 16 * hex_value[data[4]]) << 8));
        continue;

      case 4:
        if (byte_count != 2)
        {
          bfd_error_handler(abfd, dcgettext("bfd", "%pB:%u: bad extended linear address record length in Intel Hex file", 5), line_count);
          bfd_set_error(0x11);
          error = 1;
          break;
        }
        current_section = 0;
        extended_address = ((hex_value[data[3]] + 16 * hex_value[data[2]] + ((hex_value[data[1]] + 16 * hex_value[data[0]]) << 8)) << 16);
        continue;

      case 5:
        if ((byte_count - 2) & 0xFFFFFFFD)
        {
          bfd_error_handler(abfd, dcgettext("bfd", "%pB:%u: bad extended linear start address length in Intel Hex file", 5), line_count);
          bfd_set_error(0x11);
          error = 1;
          break;
        }
        v = ((hex_value[data[3]] + 16 * hex_value[data[2]] + ((hex_value[data[1]] + 16 * hex_value[data[0]]) << 8)) << 16);
        if (byte_count == 2)
        {
          abfd[21] += v;
          current_section = 0;
        }
        else
        {
          current_section = 0;
          abfd[21] = (hex_value[data[7]] + 16 * hex_value[data[6]] + ((hex_value[data[5]] + 16 * hex_value[data[4]]) << 8)) + v;
        }
        continue;

      default:
        bfd_error_handler(abfd, dcgettext("bfd", "%pB:%u: unrecognized ihex type %u in Intel Hex file", 5), line_count, rec_type);
        bfd_set_error(0x11);
        error = 1;
        break;
    }
    if (error) break;
  }

  if (!error && bfd_get_error() != 18)
    error = 1;

  if (error)
  {
    long long current = abfd[31];
    if (current != saved_section_ptr)
    {
      if (current)
        bfd_release(abfd, current);
    }
    abfd[31] = saved_section_ptr;
    if (data)
      free(data);
    return 0;
  }

  if (data)
    free(data);
  return abfd[1];
}

struct node *getuidbyname(char *src) {
    struct node *cur;
    char first_char = *src;

    cur = user_alist;
    while (cur) {
        if (cur->name[0] == first_char && strcmp(cur->name, src) == 0)
            return cur;
        cur = cur->next;
    }
    cur = nouser_alist;
    while (cur) {
        if (cur->name[0] == first_char && strcmp(cur->name, src) == 0)
            return 0;
        cur = cur->next;
    }
    size_t len = strlen(src);
    size_t alloc_size = (len + 24) & ~7;
    struct node *new_node = xmalloc(alloc_size);
    strncpy(new_node->name, src, len + 1);
    struct passwd *pw = getpwnam(src);
    if (pw) {
        new_node->uid = pw->pw_uid;
        new_node->next = user_alist;
        user_alist = new_node;
        return new_node;
    } else {
        new_node->next = nouser_alist;
        nouser_alist = new_node;
        return 0;
    }
}

        }

MagickBooleanType ImageToFile(Image *image, const char *filename, ExceptionInfo *exception)
{
  BlobInfo *blob;
  int file;
  struct stat stat_buf;
  unsigned char *buffer;
  size_t allocation_size;
  size_t bytes_read;
  ssize_t bytes_written;
  size_t total_written;
  const unsigned char *read_data;
  int status;
  int saved_errno;
  char *error_message;

  if (image == (Image *) NULL)
    return(MagickFalse);
  if (image->signature != MagickCoreSignature)
    return(MagickFalse);
  blob=image->blob;
  if (blob == (BlobInfo *) NULL)
    return(MagickFalse);
  if (blob->type == UndefinedStream)
    return(MagickFalse);
  if (image->debug != MagickFalse)
    (void) LogMagickEvent(TraceEvent,GetMagickModule(),"%s",filename);
  if (filename == (const char *) NULL)
    return(MagickFalse);
  if (*filename != '\0')
  {
    if (LocaleCompare(filename,"-") == 0)
    {
      file=fileno(stdout);
      if (file == -1)
        {
          ThrowMagickException(exception,GetMagickModule(),2340,435,
            "UnableToWriteBlob","'%s': %s",filename,
            GetExceptionMessage(errno));
          return(MagickFalse);
        }
    }
    else
      file=open(filename,O_WRONLY | O_CREAT | O_TRUNC,S_IRUSR | S_IWUSR);
  }
  else
    file=AcquireUniqueFileResource(filename);
  if (file == -1)
    {
      ThrowMagickException(exception,GetMagickModule(),2340,435,
        "UnableToWriteBlob","'%s': %s",filename,
        GetExceptionMessage(errno));
      return(MagickFalse);
    }
  if (__fxstat(_STAT_VER,file,&stat_buf) == 0)
  {
    allocation_size=(size_t) Max(stat_buf.st_size,1);
    if (allocation_size > 0x13FFF)
      allocation_size=0x14000;
  }
  else
    allocation_size=0x14000;
  buffer=(unsigned char *) AcquireQuantumMemory(allocation_size,sizeof(*buffer));
  if (buffer == (unsigned char *) NULL)
  {
    (void) close(file);
    ThrowMagickException(exception,GetMagickModule(),2350,400,
      "MemoryAllocationError","`%s'",filename);
    return(MagickFalse);
  }
  while ((bytes_read=ReadBlobStream(image,allocation_size,buffer,&read_data)) > 0)
  {
    total_written=0;
    while (total_written < bytes_read)
    {
      bytes_written=write(file,read_data+total_written,bytes_read-total_written);
      if (bytes_written <= 0)
      {
        if (errno == EINTR)
          continue;
        saved_errno=errno;
        error_message=GetExceptionMessage(saved_errno);
        (void) close(file);
        RelinquishMagickMemory(buffer);
        ThrowMagickException(exception,GetMagickModule(),2378,435,
          "UnableToWriteBlob","'%s': %s",filename,error_message);
        DestroyString(error_message);
        return(MagickFalse);
      }
      total_written+=(size_t) bytes_written;
    }
  }
  status=close(file);
  RelinquishMagickMemory(buffer);
  if (status == -1)
  {
    ThrowMagickException(exception,GetMagickModule(),2378,435,
      "UnableToWriteBlob","'%s': %s",filename,GetExceptionMessage(errno));
    return(MagickFalse);
  }
  return(MagickTrue);
}

void add_regexp(struct regexp_buffer *buf, const char *pattern)
{
    size_t pattern_len;
    char *err;
    char *old_data;
    size_t old_len;
    size_t new_len;
    char *new_data;
    int need_sep;

    if (buf == NULL || pattern == NULL)
        return;

    pattern_len = strlen(pattern);
    err = rpl_re_compile_pattern(pattern, pattern_len, buf->preg);
    if (err)
    {
        error(2, 0, "%s: %s", pattern, err);
        return;
    }

    old_data = buf->data;
    old_len = buf->len;
    need_sep = (old_data != NULL);
    buf->need_separator = need_sep;
    new_len = old_len + pattern_len + (need_sep ? 2 : 0);

    if (new_len > buf->capacity)
    {
        size_t new_cap = buf->capacity;
        if (new_cap == 0)
            new_cap = 1;
        while (new_len > new_cap)
            new_cap *= 2;
        new_data = xrealloc(old_data, new_cap);
        buf->data = new_data;
        buf->capacity = new_cap;
    }
    else
    {
        new_data = buf->data;
    }

    buf->len = new_len;

    if (old_data != NULL)
    {
        new_data[old_len] = '\\';
        new_data[old_len + 1] = '|';
    }

    memcpy(new_data + old_len + (need_sep ? 2 : 0), pattern, pattern_len + 1);
}

void mpz_goetgheluck_bin_uiui(mpz_ptr r, mp_limb_t n, mp_limb_t k)
{
    mp_limb_t diff = ((n - 5) | 1);
    mp_limb_t sieve_size = 8 * (diff / 0xC0) + 8;
    void *tmp_marker = 0;
    mp_limb_t *sieve;
    if (sieve_size > 0x7F00)
        sieve = (mp_limb_t *)__gmp_tmp_reentrant_alloc(&tmp_marker, sieve_size);
    else
        sieve = (mp_limb_t *)alloca(16 * ((sieve_size + 30) >> 4));
    mp_limb_t prime_count = __gmp_primesieve(sieve, n) + 1;
    mp_limb_t v6 = 8;
    while (n > _gmp_limbroots_table[v6 - 1])
        v6--;
    mp_limb_t *prime_products;
    mp_limb_t prod_size = 8 * (prime_count / v6) + 23;
    if (prod_size > 0x7F00)
        prime_products = (mp_limb_t *)__gmp_tmp_reentrant_alloc(&tmp_marker, prod_size);
    else
        prime_products = (mp_limb_t *)alloca(prod_size);
    mp_limb_t product_count = 0;
    mp_limb_t diff_nk = n - k;
    mp_limb_t inv_n = ULONG_MAX / n;
    mp_limb_t t = diff_nk;
    t -= (t >> 1) & 0x5555555555555555ULL;
    t = ((t >> 2) & 0x3333333333333333ULL) + (t & 0x3333333333333333ULL);
    t = (t + (t >> 4)) & 0x0F0F0F0F0F0F0F0FULL;
    t += t >> 8;
    t += t >> 16;
    mp_limb_t popcount_nk = t & 0xFF;
    t = k;
    t -= (t >> 1) & 0x5555555555555555ULL;
    t = ((t >> 2) & 0x3333333333333333ULL) + (t & 0x3333333333333333ULL);
    t = (t + (t >> 4)) & 0x0F0F0F0F0F0F0F0FULL;
    t += t >> 8;
    t += t >> 16;
    mp_limb_t popcount_k = t & 0xFF;
    t = n;
    t -= (t >> 1) & 0x5555555555555555ULL;
    t = ((t >> 2) & 0x3333333333333333ULL) + (t & 0x3333333333333333ULL);
    t = (t + (t >> 4)) & 0x0F0F0F0F0F0F0F0FULL;
    t += t >> 8;
    t += t >> 16;
    mp_limb_t popcount_n = t & 0xFF;
    mp_limb_t product = 1ULL << (popcount_nk - (popcount_n - popcount_k));
    if (inv_n < product)
    {
        prime_products[product_count++] = product;
        product = 1;
    }
    mp_limb_t n_tmp = n, k_tmp = k;
    mp_limb_t carry = 0;
    do
    {
        mp_limb_t k_digit = k_tmp % 3;
        k_tmp /= 3;
        mp_limb_t n_digit = n_tmp % 3;
        n_tmp /= 3;
        if (n_digit < k_digit + carry)
        {
            product *= 3;
            carry = 1;
        }
        else
            carry = 0;
    }
    while (n_tmp > 2);
    unsigned long bit_idx;
    _BitScanReverse64(&bit_idx, n);
    mp_limb_t limit1 = ((((1ULL << ((int)(64 - (bit_idx ^ 0x3F)) >> 1)) + (n >> ((int)(64 - (bit_idx ^ 0x3F)) >> 1))) >> 1) - 5) | 1;
    limit1 /= 3;
    mp_limb_t mask = 1;
    mp_limb_t word_idx = 0;
    mp_limb_t i;
    for (i = 0; i <= limit1; i++)
    {
        mp_limb_t p = 3 * (i + 1) + ((i + 1) & 1) + 1;
        if ((mask & sieve[word_idx]) == 0)
        {
            if (product > inv_n)
            {
                prime_products[product_count++] = product;
                product = 1;
            }
            mp_limb_t n_t = n, k_t = k, carry2 = 0;
            do
            {
                mp_limb_t kd = k_t % p;
                k_t /= p;
                mp_limb_t nd = n_t % p;
                n_t /= p;
                if (nd < kd + carry2)
                {
                    product *= p;
                    carry2 = 1;
                }
                else
                    carry2 = 0;
            }
            while (p <= n_t);
        }
        mask = (mask << 1) | (mask >> 63);
        word_idx += (mask & 1);
    }
    mp_limb_t j = i;
    mp_limb_t p_base = 3 * i + 6;
    mp_limb_t limit2 = ((diff_nk - 5) | 1) / 3;
    for (; j <= limit2; )
    {
        j++;
        mp_limb_t p = p_base + (j & 1) + 1;
        if ((mask & sieve[word_idx]) == 0)
        {
            if (n % p < k % p)
            {
                if (product <= 2 * inv_n)
                    product *= p;
                else
                {
                    prime_products[product_count++] = product;
                    product = p;
                }
            }
        }
        mask = (mask << 1) | (mask >> 63);
        p_base += 3;
        word_idx += (mask & 1);
    }
    mp_limb_t limit3 = (diff / 3) - 1;
    mp_limb_t nk_limit = ((diff_nk - 5) | 1) / 3;
    mp_limb_t p_base2 = 3 * (((diff_nk - 5) | 1) / 3) + 6;
    mp_limb_t word_idx2 = limit3 >> 6;
    mp_limb_t mask2 = 1ULL << (limit3 & 63);
    for (; j <= limit3; )
    {
        j++;
        if ((mask2 & sieve[word_idx2]) == 0)
        {
            if (product > inv_n & 0x7FFFFFFFFFFFFFFFULL)
            {
                prime_products[product_count++] = product;
                product = p_base2 + (j & 1) + 1;
            }
            else
            {
                product *= p_base2 + (j & 1) + 1;
            }
        }
        mask2 = (mask2 << 1) | (mask2 >> 63);
        p_base2 += 3;
        word_idx2 += (mask2 & 1);
    }
    if (product_count)
    {
        prime_products[product_count] = product;
        __gmpz_prodlimbs(r, prime_products, product_count + 1);
    }
    else
    {
        mp_limb_t alloc = r->_mp_alloc;
        if ((int)alloc <= 0)
            r->_mp_d = (mp_limb_t *)__gmpz_realloc(r, 1);
        r->_mp_d[0] = product;
        r->_mp_size = 1;
    }
    if (tmp_marker)
        __gmp_tmp_reentrant_free(tmp_marker);
}
