        // popcount of n
        x = n;
        x = x - ((x >> 1) & 0x5555555555555555ULL);
        x = (x & 0x3333333333333333ULL) + ((x >> 2) & 0x3333333333333333ULL);
        x = (x + (x >> 4)) & 0x0F0F0F0F0F0F0F0FULL;
        x = x + (x >> 8);
        x = x + (x >> 16);
        x = x + (x >> 32);
        mp_limb_t pop_n = x & 0xFF;

        mp_limb_t exponent2 = pop_n_minus_k + pop_k - pop_n;
        product = (mp_limb_t)1 << exponent2;
    }

    // Handle prime 3 using base-3 representation
    {
        mp_limb_t a = k, b = n;
        int carry = 0;
        do
        {
            mp_limb_t qa = a / 3, ra = a % 3 + carry;
            mp_limb_t qb = b / 3, rb = b % 3;
            carry = (ra > rb) ? 1 : 0;
            if (carry)
                product *= 3;
            a = qa;
            b = qb;
        } while (b > 2);
    }

    // Determine limit for prime sieve loop
    mp_limb_t sqrt_n;
    {
        unsigned long bit;
        _BitScanReverse64(&bit, n);
        sqrt_n = ((((mp_limb_t)1 << ((bit + 1) >> 1)) + (n >> ((bit + 1) >> 1))) >> 1);
        sqrt_n = ((sqrt_n - 5) | 1) / 3;
    }

    // Main loop over primes up to sqrt(n)
    {
        mp_limb_t mask = 1;
        mp_size_t idx = 0;
        mp_limb_t p_index = 0;
        mp_limb_t a = k, b = n;
        mp_limb_t current_product = product;
        mp_limb_t factor_list[1000]; // placeholder, actual size determined earlier

        while (1)
        {
            p_index++;
            if (!(sieve[idx] & mask))
            {
                mp_limb_t p = 3 * p_index + (p_index & 1) + 1;
                if (current_product > inv_n)
                {
                    factors[factor_count++] = current_product;
                    current_product = 1;
                }
                mp_limb_t carry = 0;
                mp_limb_t aa = a, bb = b;
                do
                {
                    mp_limb_t qa = aa / p, ra = aa % p + carry;
                    mp_limb_t qb = bb / p, rb = bb % p;
                    carry = (ra > rb) ? 1 : 0;
                    if (carry)
                        current_product *= p;
                    aa = qa;
                    bb = qb;
                } while (bb >= p);
            }
            mask = (mask << 1) | (mask >> 63);
            idx += (mask & 1);
            if (p_index > sqrt_n)
                break;
        }
        product = current_product;
    }

    // Second loop for primes between sqrt(n) and n
    {
        mp_limb_t p = 3 * (p_index + 1) + 6; // starting prime after sqrt_n
        mp_limb_t mask = 1;
        mp_size_t idx = 0;
        mp_limb_t limit = ((n - 5) | 1) / 3;
        mp_limb_t a = n, b = k;
        while (1)
        {
            p_index++;
            if (!(sieve[idx] & mask))
            {
                mp_limb_t q = p + (p_index & 1) + 1;
                if (a % q < b % q)
                {
                    if (product <= inv_n * 2)
                        product *= q;
                    else
                    {
                        factors[factor_count++] = product;
                        product = q;
                    }
                }
            }
            mask = (mask << 1) | (mask >> 63);
            idx += (mask & 1);
            p += 3;
            if (p_index > limit)
                break;
        }
    }

    // Third loop for primes between n/2 and n
    {
        mp_limb_t limit2 = ((n_minus_k - 5) | 1) / 3;
        mp_limb_t p = 3 * (limit2 + 1) + 6;
        mp_limb_t mask = 1;
        mp_size_t idx = 0;
        mp_limb_t a = n, b = k;
        while (1)
        {
            p_index++;
            if (!(sieve[idx] & mask))
            {
                mp_limb_t q = p + (p_index & 1) + 1;
                if (product <= inv_n)
                    product *= q;
                else
                {
                    factors[factor_count++] = product;
                    product = q;
                }
            }
            mask = (mask << 1) | (mask >> 63);
            idx += (mask & 1);
            p += 3;
            if (p_index > limit2)
                break;
        }
    }

    // Finalize product
    if (factor_count)
    {
        factors[factor_count] = product;
        __gmpz_prodlimbs(r, factors, factor_count + 1);
    }
    else
    {
        if (r->_mp_size <= 0)
        {
            r->_mp_d = (mp_limb_t *)__gmpz_realloc(r, 1);
        }
        r->_mp_d[0] = product;
        r->_mp_size = 1;
    }

    if (marker)
        __gmp_tmp_reentrant_free(marker);
}
