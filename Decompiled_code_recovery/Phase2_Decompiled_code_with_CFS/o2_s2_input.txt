    if (bfd_bread(header, 9, abfd) != 9)
    {
        ret = bfd_get_error();
        if (ret == 18)
        {
            bfd_set_error(bfd_error_file_not_recognized);
        }
        return FALSE;
    }

    if (header[0] != ':')
        goto not_ihex;

    i = 1;
    while (i < 9)
    {
        if (hex_value[(unsigned char)header[i]] == 99)
            goto not_ihex;
        i++;
    }

    if ((unsigned char)hex_value[(unsigned char)header[8]] + 16 * (unsigned char)hex_value[(unsigned char)header[7]] <= 5)
        goto not_ihex;

    {
        asection *old_sections = abfd->sections;
        asection *new_section = bfd_alloc(abfd, sizeof(asection));
        if (!new_section)
        {
            abfd->sections = old_sections;
            return FALSE;
        }
        abfd->sections = new_section;
        new_section->next = NULL;
        new_section->prev = NULL;

        if (bfd_seek(abfd, 0, SEEK_SET) != 0)
        {
            bfd_release(abfd, new_section);
            abfd->sections = old_sections;
            return FALSE;
        }

        abfd->start_address = 0;
        lineno = 1;
        data_buf = NULL;
        bufsize = 0;
        current_section = NULL;
        base = 0;
        seg = 0;

        while (bfd_bread(&byte, 1, abfd) == 1)
        {
            int ch = byte;
            if (ch == '\r')
                continue;
            if (ch == '\n')
            {
                lineno++;
                continue;
            }
            if (ch != ':')
            {
                ihex_bad_byte(abfd, lineno, ch, 0);
                goto error_cleanup;
            }

            bfd_vma filepos = bfd_tell(abfd);
            if (bfd_bread(hexbuf, 8, abfd) != 8)
                goto error_cleanup;

            for (int j = 0; j < 8; j++)
            {
                if (hex_value[hexbuf[j]] == 99)
                {
                    ihex_bad_byte(abfd, lineno, hexbuf[j], 0);
                    goto error_cleanup;
                }
            }

            byte_count = hex_value[hexbuf[1]] + 16 * hex_value[hexbuf[0]];
            len = 2 * byte_count + 2;

            unsigned char h0 = hex_value[hexbuf[0]];
            unsigned char h1 = hex_value[hexbuf[1]];
            unsigned char h2 = hex_value[hexbuf[2]];
            unsigned char h3 = hex_value[hexbuf[3]];
            unsigned char h4 = hex_value[hexbuf[4]];
            unsigned char h5 = hex_value[hexbuf[5]];
            unsigned char h6 = hex_value[hexbuf[6]];
            unsigned char h7 = hex_value[hexbuf[7]];

            bfd_vma address = (h1 + 16 * h0) << 8;
            address += h2 + 16 * h3;
            unsigned int record_type = h4 + 16 * h5;
            unsigned char expected_csum = h6 + 16 * h7;
            int byte_count2 = byte_count;

            if (len > bufsize)
            {
                unsigned char *new_buf = bfd_realloc(data_buf, len);
                if (!new_buf)
                    goto error_cleanup;
                data_buf = new_buf;
                bufsize = len;
            }

            if (bfd_bread(data_buf, len, abfd) != len)
                goto error_cleanup;

            for (int j = 0; j < len; j++)
            {
                if (hex_value[data_buf[j]] == 99)
                {
                    ihex_bad_byte(abfd, lineno, data_buf[j], 0);
                    goto error_cleanup;
                }
            }

            int checksum = byte_count + (address >> 8) + (address & 0xFF) + record_type;
            for (int j = 0; j < 2 * byte_count; j += 2)
            {
                checksum += hex_value[data_buf[j + 1]] + 16 * hex_value[data_buf[j]];
            }
            checksum = (unsigned char)(-checksum);
            if (checksum != expected_csum)
            {
                bfd_error_handler("%pB:%u: bad checksum in Intel Hex file (expected %u, found %u)",
                                  abfd, lineno, checksum, expected_csum);
                bfd_set_error(bfd_error_file_format_not_recognized);
                goto error_cleanup;
            }

            switch (record_type)
            {
                case 0:
                    if (current_section && (current_section->vma + current_section->size == base + seg + address))
                    {
                        current_section->size += byte_count;
                    }
                    else if (byte_count > 0)
                    {
                        snprintf(name, sizeof(name), ".sec%u", lineno);
                        char *name_copy = bfd_alloc(abfd, strlen(name) + 1);
                        if (!name_copy)
                            goto error_cleanup;
                        strcpy(name_copy, name);
                        current_section = bfd_make_section_with_flags(abfd, name_copy, SEC_HAS_CONTENTS | SEC_ALLOC | SEC_LOAD);
                        if (!current_section)
                            goto error_cleanup;
                        bfd_vma section_address = base + seg + address;
                        current_section->vma = section_address;
                        current_section->lma = section_address;
                        current_section->size = byte_count;
                        current_section->filepos = filepos - 1;
                    }
                    break;

                case 1:
                    if (!abfd->start_address)
                        abfd->start_address = address;
                    break;

                case 2:
                    if (byte_count != 2)
                    {
                        bfd_error_handler("%pB:%u: bad extended address record length in Intel Hex file", abfd, lineno);
                        bfd_set_error(bfd_error_file_format_not_recognized);
                        goto error_cleanup;
                    }
                    current_section = NULL;
                    seg = 16 * (hex_value[data_buf[3]] + 16 * hex_value[data_buf[2]] +
                               ((hex_value[data_buf[1]] + 16 * hex_value[data_buf[0]]) << 8));
                    break;

                case 3:
                    if (byte_count != 4)
                    {
                        bfd_error_handler("%pB:%u: bad extended start address length in Intel Hex file", abfd, lineno);
                        bfd_set_error(bfd_error_file_format_not_recognized);
                        goto error_cleanup;
                    }
                    current_section = NULL;
                    abfd->start_address += 16 * (hex_value[data_buf[3]] + 16 * hex_value[data_buf[2]] +
                                                  ((hex_value[data_buf[1]] + 16 * hex_value[data_buf[0]]) << 8))
                                          + hex_value[data_buf[7]] + 16 * hex_value[data_buf[6]]
                                          + ((hex_value[data_buf[5]] + 16 * hex_value[data_buf[4]]) << 8);
                    break;

                case 4:
                    if (byte_count != 2)
                    {
                        bfd_error_handler("%pB:%u: bad extended linear address record length in Intel Hex file", abfd, lineno);
                        bfd_set_error(bfd_error_file_format_not_recognized);
                        goto error_cleanup;
                    }
                    current_section = NULL;
                    base = ((hex_value[data_buf[3]] + 16 * hex_value[data_buf[2]] +
                            ((hex_value[data_buf[1]] + 16 * hex_value[data_buf[0]]) << 8)) << 16);
                    break;

                case 5:
                    if (byte_count != 2 && byte_count != 4)
                    {
                        bfd_error_handler("%pB:%u: bad extended linear start address length in Intel Hex file", abfd, lineno);
                        bfd_set_error(bfd_error_file_format_not_recognized);
                        goto error_cleanup;
                    }
                    bfd_vma linear_addr = ((hex_value[data_buf[3]] + 16 * hex_value[data_buf[2]] +
                                            ((hex_value[data_buf[1]] + 16 * hex_value[data_buf[0]]) << 8)) << 16);
                    current_section = NULL;
                    if (byte_count == 2)
                        abfd->start_address += linear_addr;
                    else
                        abfd->start_address = hex_value[data_buf[7]] + 16 * hex_value[data_buf[6]]
                                             + ((hex_value[data_buf[5]] + 16 * hex_value[data_buf[4]]) << 8)
                                             + linear_addr;
                    break;

                default:
                    bfd_error_handler("%pB:%u: unrecognized ihex type %u in Intel Hex file", abfd, lineno, record_type);
                    bfd_set_error(bfd_error_file_format_not_recognized);
                    goto error_cleanup;
            }
        }

        if (bfd_get_error() != 18)
            goto error_cleanup;

        if (data_buf)
            free(data_buf);
        return TRUE;

    error_cleanup:
        if (data_buf)
            free(data_buf);
        if (abfd->sections)
            bfd_release(abfd, abfd->sections);
        abfd->sections = old_sections;
        return FALSE;
    }

not_ihex:
    if (bfd_get_error() == 18)
        bfd_set_error(bfd_error_file_not_recognized);
    return FALSE;
}
